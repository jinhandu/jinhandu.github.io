Exercise1 arrayFromRange

function arrayFromRange(min,max){
    let array=[];
    for (let value =min;value<=max;value++){
        array.push(value);
    }
    return array;
}
const arr=arrayFromRange(-8,10);

Esercise2 Includes

function myIncludes (array,searchElement){
    for (let value of array) {
        if (searchElement===value) return true;
    }
    return false;
}

const arr=[1,2,3,4,5,123,6];
console.log(myIncludes(arr,28));

Exercise3 except

function except(originalArray,removeArray){
    for (let removeValue of removeArray){
        while (originalArray.includes(removeValue)){
            originalArray.splice(originalArray.indexOf(removeValue),1);
        };
    }
    return originalArray;
}

const a =[1,2,3,4,5,1,1,1,1,1,1,1,3,5,6,78,];
console.log(except(a,[1,3]));

Exercise4 move

function move(array,index,offset){
    const output=[...array];
    const insertIndex=index+offset;
    
    if (insertIndex<0||insertIndex>output.length-1){
        console.error("Invalid offset");
        return;
    };

    const moveElement=output[index];

    //remove
    output.splice(index,1);
    output.splice(insertIndex,0,moveElement);

    return output;
}
let array=[0,1,2,3,4,5];
console.log(move(array,1,5));

function countOccurrences(array,searchElement){
    const temArray=[...array];
    let counter=0;
    while (temArray.includes(searchElement)){
        temArray.splice(temArray.indexOf(searchElement),1);
        counter++; 
    }
    return counter;
}

    const temArray=[...array];
    let counter=0;
    temArray.reduce(function(a,currentValue){
        a=currentValue;
        if (searchElement==a){
            counter++;
        }
    },null)
    return counter;
}

Exercise6 getMax

function getMax(array){
    if (array.length===0) return undefined;
    let maxNumber;
    for (let number of array){
        if (maxNumber===undefined||number > maxNumber)maxNumber=number;
    }
    return maxNumber;
}
console.log(getMax([1,2,3,4,5,6]));
