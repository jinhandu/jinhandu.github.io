Exercise3 fizzbuzz algorithm

// Divisible by 3 =>Fizz
// Divisible by 5 =>Buzz
// Divisible by both 3 and 5 =>FizzBuzz
// Not divisible by 3 or 5 => input
// Not a number ==> 'Not a number'

console.log(fizzBuzz(250));

function fizzBuzz(input) {
    let inputType = typeof(input);
    if (inputType !== 'number') return NaN;
    if (input % 3 === 0 && input % 5 === 0) return 'FizzBuzz';
    else if (input % 3 === 0) return 'Fizz';
    else if (input % 5 === 0) return 'Buzz';
    else return input;
}
  • 细节⇒ NaN 的意思是 Not a Number 但是NaN 本身的typeof是number

Exercise4 checkSpeed

//Speed Limit = 70 OK
// 74 => OK
// 5 -> 1 point
// 12 points -> License suspended

//Tips:
// Math.floor(1.3)

console.log(checkSpeed(74));

function checkSpeed(speed) {
    const speedLimit = 70;
    const kmPerPoint = 5;
    let inputType = typeof(speed);
    if (inputType !== 'number') return NaN;
    if (speed < speedLimit + kmPerPoint) return 'OK';
    const maxpoints = 12;

    const points = Math.floor((speed - speedLimit) / kmPerPoint);
    if (points >= maxpoints) return 'License suspended';
    return 'points:' + points;

}
  • 默认情况下 尽量写const 如果需要改变 写let
  • 数值尽量存到 变量里面 不要写在判断条件里

function countTruthy(ary) {
    //const falsyValues = [undefined, null, '', false, 0, NaN];
    let counter = 0;

    for (let value of ary) {
        if (value) counter++;
    }
    return counter;
}

function showProperties(obj) {
    var showObj = {};
    for (key in obj) {
        if (typeof(obj[key]) === 'string') {
            showObj[key] = obj[key];
        }
    }
    return showObj;

Exercise 9 calculateGrade

const ary = [100, 100, 40];
calculateGrade(ary);

function calculateGrade(ary) {
    let sum = 0;
    let average = 0;
    let mark;
    for (aryValue of ary) {
        sum += aryValue;
    }
    average = sum / ary.length;
    if (average < 60) {
        mark = 'E';
    } else if (average < 70) mark = 'D';
    else if (average < 80) mark = 'C';
    else if (average < 90) mark = 'B';
    else if (average < 100) mark = 'A';

    console.log(average, mark);
}
  • single responsible principle 每个函数只负责一个事 所以可以把求平均值单独写出来
