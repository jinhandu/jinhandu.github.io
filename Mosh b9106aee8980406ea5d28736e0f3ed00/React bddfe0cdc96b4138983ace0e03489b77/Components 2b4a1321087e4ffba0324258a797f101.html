Setting Attributes

import React, { Component } from "react";

class Counter extends Component {
  state = {
    count: 0,
  };
  style = {
    fontSize: 50,
    fontWeight: "bold",
  };
  render() {
    return (
      <React.Fragment>
        <h1 style={this.style} className="badge badge-primary m-2">
          {" "}
          {this.formatCount()}
        </h1>
        <button>Increment</button>
      </React.Fragment>
    );
  }
  formatCount() {
    const { count } = this.state;
    return count === 0 ? "Zero" : count;
  }
}

export default Counter;

Rendering Classes Dynamically

import React, { Component } from "react";

class Counter extends Component {
  state = {
    count: 0,
  };
  style = {
    fontSize: 30,
    fontWeight: "bold",
  };
  render() {
    return (
      <React.Fragment>
        <h1 style={this.style} className={this.getBadgeClasses()}>
          {" "}
          {this.formatCount()}
        </h1>
        <button className="btn btn-secondary btn-sm">Increment</button>
      </React.Fragment>
    );
  }
  getBadgeClasses() {
    let classes = "badge m-2 badge-";
    classes += this.state.count === 0 ? "warning" : "primary";
    return classes;
  }

  formatCount() {
    const { count } = this.state;
    return count === 0 ? "Zero" : count;
  }
}

export default Counter;
